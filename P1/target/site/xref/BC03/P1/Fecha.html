4   
5   	@SuppressWarnings("serial")
6   	static class DatoInvalidoException extends Exception {

9   		}
10  	}

12  	private int dia, mes, agno;
13  
14  	public Fecha(int dia, int mes, int agno) throws DatoInvalidoException {
15  		this.setDia(dia);
16  		this.setMes(mes);
17  		this.setAgno(agno);
18  	}
19  
20  	public void setDia(int dia) throws DatoInvalidoException {
21  		if (dia >= 1 && dia <= 31)
22  			this.dia = dia;
23  		else
24  			throw new DatoInvalidoException("El día debe ser un numero entre 1 y 31");
25  	}
26  
27  	public void setMes(int mes) throws DatoInvalidoException {
28  		if (mes >= 1 && mes <= 12)
29  			this.mes = mes;
30  		else
31  			throw new DatoInvalidoException("El mes debe ser un numero entre 1 y 12");
32  	}
33  
34  	public void setAgno(int agno) throws DatoInvalidoException {
35  		if (agno >= 1)
36  			this.agno = agno;
37  		else
38  			throw new DatoInvalidoException("El año debe ser un numero positivo");
39  	}
40  
41  	public boolean comprobarBisiesto() {
42  		boolean bisiesto = this.agno % 4 == 0 && this.agno % 100 != 0 || this.agno % 400 == 0;
43  		return bisiesto;
44  	}
